Section 31: Selenium + Cucumber BDD

Why Cucumber with Selenium?

Behavior-Driven Development (BDD) bridges the gap between business stakeholders and automation engineers. Cucumber reads plain-English Gherkin scenarios and maps them to Selenium step definitions, making tests readable by product owners, BA, and QA alike.

  • Business-readable: Scenarios describe user behavior, not Java syntax.
  • Collaboration: Three Amigos (BA, Dev, QA) define acceptance criteria together.
  • Living documentation: Feature files stay in sync with the application and serve as executable specs.
Key Idea: Gherkin describes what the user does; step definitions describe how Selenium performs it.

1. login.feature — 4 Scenarios

Feature: User Login
  As a registered user
  I want to log in to the application
  So that I can access my dashboard

  Background:
    Given the user is on the login page

  @smoke @positive
  Scenario: Successful login with valid credentials
    When the user enters username "john.doe@example.com" and password "SecurePass123"
    And clicks the login button
    Then the user should see the dashboard
    And the welcome message should contain "Welcome, John"

  @negative
  Scenario: Login fails with invalid password
    When the user enters username "john.doe@example.com" and password "wrongpass"
    And clicks the login button
    Then an error message "Invalid credentials" should be displayed

  @negative
  Scenario Outline: Login fails with invalid data
    When the user enters username "<username>" and password "<password>"
    And clicks the login button
    Then an error message "<error>" should be displayed

    Examples:
      | username              | password    | error                  |
      |                       | SecurePass  | Username is required   |
      | john.doe@example.com  |             | Password is required   |
      | invalid-email         | SecurePass  | Invalid email format   |

  @regression
  Scenario: Remember me checkbox persists session
    When the user enters username "john.doe@example.com" and password "SecurePass123"
    And checks the remember me checkbox
    And clicks the login button
    Then the user should see the dashboard
    And the session cookie should be present

2. LoginPage.java — Page Object

package pages;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.FindBy;
import org.openqa.selenium.support.PageFactory;

public class LoginPage {
    private WebDriver driver;

    @FindBy(id = "username")
    private WebElement usernameField;

    @FindBy(id = "password")
    private WebElement passwordField;

    @FindBy(id = "login-btn")
    private WebElement loginButton;

    @FindBy(id = "remember-me")
    private WebElement rememberMeCheckbox;

    @FindBy(css = ".error-message")
    private WebElement errorMessage;

    public LoginPage(WebDriver driver) {
        this.driver = driver;
        PageFactory.initElements(driver, this);
    }

    public void navigateTo(String url) {
        driver.get(url);
    }

    public void enterUsername(String username) {
        usernameField.clear();
        usernameField.sendKeys(username);
    }

    public void enterPassword(String password) {
        passwordField.clear();
        passwordField.sendKeys(password);
    }

    public void clickLogin() {
        loginButton.click();
    }

    public void checkRememberMe() {
        if (!rememberMeCheckbox.isSelected()) {
            rememberMeCheckbox.click();
        }
    }

    public String getErrorMessage() {
        return errorMessage.getText();
    }

    public boolean isErrorDisplayed() {
        return errorMessage.isDisplayed();
    }
}

3. LoginSteps.java — Step Definitions

package stepdefinitions;

import context.TestContext;
import io.cucumber.java.en.*;
import org.testng.Assert;
import pages.DashboardPage;
import pages.LoginPage;

public class LoginSteps {
    private TestContext context;
    private LoginPage loginPage;
    private DashboardPage dashboardPage;

    public LoginSteps(TestContext context) {
        this.context = context;
    }

    @Given("the user is on the login page")
    public void userIsOnLoginPage() {
        loginPage = new LoginPage(context.getDriver());
        loginPage.navigateTo(context.getBaseUrl() + "/login");
    }

    @When("the user enters username {string} and password {string}")
    public void enterCredentials(String username, String password) {
        loginPage.enterUsername(username);
        loginPage.enterPassword(password);
    }

    @When("clicks the login button")
    public void clickLoginButton() {
        loginPage.clickLogin();
    }

    @When("checks the remember me checkbox")
    public void checkRememberMe() {
        loginPage.checkRememberMe();
    }

    @Then("the user should see the dashboard")
    public void verifyDashboard() {
        dashboardPage = new DashboardPage(context.getDriver());
        Assert.assertTrue(dashboardPage.isDashboardDisplayed());
    }

    @Then("the welcome message should contain {string}")
    public void verifyWelcomeMessage(String expected) {
        Assert.assertTrue(dashboardPage.getWelcomeMessage().contains(expected));
    }

    @Then("an error message {string} should be displayed")
    public void verifyErrorMessage(String expected) {
        Assert.assertTrue(loginPage.isErrorDisplayed());
        Assert.assertEquals(loginPage.getErrorMessage(), expected);
    }

    @Then("the session cookie should be present")
    public void verifySessionCookie() {
        Assert.assertNotNull(context.getDriver().manage().getCookieNamed("SESSIONID"));
    }
}

4. Hooks.java — @Before / @After

package hooks;

import context.TestContext;
import io.cucumber.java.After;
import io.cucumber.java.Before;
import io.cucumber.java.Scenario;
import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import java.time.Duration;

public class Hooks {
    private TestContext context;

    public Hooks(TestContext context) {
        this.context = context;
    }

    @Before
    public void setUp(Scenario scenario) {
        ChromeOptions options = new ChromeOptions();
        options.addArguments("--start-maximized");
        WebDriver driver = new ChromeDriver(options);
        driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(10));
        context.setDriver(driver);
        context.setBaseUrl("https://app.example.com");
    }

    @After
    public void tearDown(Scenario scenario) {
        if (scenario.isFailed()) {
            byte[] screenshot = ((TakesScreenshot) context.getDriver())
                .getScreenshotAs(OutputType.BYTES);
            scenario.attach(screenshot, "image/png", scenario.getName());
        }
        if (context.getDriver() != null) {
            context.getDriver().quit();
        }
    }
}

Sharing State: TestContext / ScenarioContext Pattern

Cucumber creates a new step definition instance per scenario by default. Use a shared context object injected via PicoContainer or manual constructor injection to pass WebDriver and test data between steps.

package context;

import org.openqa.selenium.WebDriver;

public class TestContext {
    private WebDriver driver;
    private String baseUrl;

    public WebDriver getDriver() { return driver; }
    public void setDriver(WebDriver driver) { this.driver = driver; }
    public String getBaseUrl() { return baseUrl; }
    public void setBaseUrl(String baseUrl) { this.baseUrl = baseUrl; }
}
Dependency Injection: Add cucumber-picocontainer to pom.xml. Cucumber auto-wires classes with shared constructor parameters.

Cucumber + Selenium Best Practices

  1. Keep steps reusable: One step = one action. Avoid scenario-specific step text.
  2. Use Page Objects: Step definitions should delegate to POM, not contain locators.
  3. Tag strategically: @smoke, @regression, @wip for selective execution.
  4. Avoid technical language in Gherkin: Write "clicks login" not "clicks element by id login-btn".
  5. One scenario = one behavior: Do not chain unrelated validations in a single scenario.
  6. Use Background for common setup: Reduce duplication across scenarios in a feature.
  7. Attach screenshots on failure: Use @After hook with scenario.attach().
  8. Parallel execution carefully: Use ThreadLocal WebDriver with PicoContainer scoped per scenario.

BDD Anti-Patterns with Selenium

Anti-PatternProblemFix
Imperative Gherkin"Given I open Chrome, When I find element by xpath..."Describe behavior, not implementation
God Step DefinitionsOne step does login + navigation + assertionSplit into atomic reusable steps
Duplicate step textSame action with 5 different step phrasesConsolidate to one canonical step
Locators in feature filesXPath in GherkinKeep locators in Page Objects only
Testing through UI onlyUI setup for data that API can createUse API for test data, UI for validation
Overloaded Examples tables50-row Scenario OutlineUse DataProvider or external CSV for bulk data

What Is Maven?

Apache Maven is a build automation and dependency management tool for Java projects. It uses a Project Object Model (POM) to define project structure, libraries, plugins, and build lifecycle — eliminating manual JAR management.

  • Build automation: Compile, test, package, and deploy with standardized commands.
  • Dependency management: Auto-downloads transitive dependencies from Maven Central.
  • Convention over configuration: Standard directory layout (src/main/java, src/test/java).

│       └── resources/      # config files, testng.xml, log4j2.xml

Maven Surefire Plugin

Surefire executes unit and integration tests during the test phase. Key configurations:

  • testng.xml integration: <suiteXmlFiles> points to TestNG suite
  • Parallel execution: <parallel>methods</parallel> with <threadCount>3</threadCount>
  • Includes/excludes: <includes><include>**/*Test.java</include></includes>

Section 33: Selenium + Jenkins CI/CD

CI/CD Pipeline for Selenium

Continuous Integration automatically builds and tests code on every commit. Continuous Delivery extends this to deploy validated builds. For Selenium, CI/CD ensures regression runs on schedule and after every push.

Developer Push
      │
      ▼
   GitHub (Webhook)
      │
      ▼
   Jenkins Job Triggered
      │
      ├── Checkout Code
      ├── Maven Build (mvn clean compile)
      ├── Run Tests (mvn test -Dsuite=smoke.xml)
      ├── Generate Reports (Extent / Allure / TestNG)
      └── Notification (Email / Slack)
            │
            ▼
      Pass ✅  or  Fail ❌

Jenkins Job Configuration

1. Freestyle Project — Step-by-Step

  1. Create New Item → Freestyle project → name: Selenium-Regression
  2. Source Code Management → Git → Repository URL + credentials
  3. Build Triggers → Poll SCM or GitHub hook trigger
  4. Build Environment → Delete workspace before build (optional)
  5. Build Steps → Invoke top-level Maven targets: clean test -Dbrowser=chrome -Denv=qa
  6. Post-build Actions → Publish TestNG Results, Publish HTML reports, Email notification

2. Pipeline Project — Jenkinsfile (Preferred)

Pipeline-as-Code version-controls the entire CI/CD definition alongside test code. Supports parameters, parallel stages, and conditional logic.

Complete Jenkinsfile

pipeline {
    agent any

    parameters {
        choice(name: 'BROWSER', choices: ['chrome', 'firefox', 'edge'], description: 'Browser to run tests')
        choice(name: 'ENVIRONMENT', choices: ['qa', 'staging', 'prod'], description: 'Target environment')
        choice(name: 'SUITE', choices: ['smoke', 'regression', 'full'], description: 'Test suite to execute')
    }

    environment {
        MAVEN_HOME = tool 'Maven-3.9'
        JAVA_HOME = tool 'JDK-17'
        PATH = "${MAVEN_HOME}/bin:${JAVA_HOME}/bin:${PATH}"
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/myorg/selenium-framework.git'
            }
        }
        stage('Build') {
            steps {
                bat 'mvn clean compile -DskipTests'
            }
        }
        stage('Test') {
            steps {
                bat "mvn test -Dbrowser=${params.BROWSER} -Denv=${params.ENVIRONMENT} -Dsurefire.suiteXmlFiles=src/test/resources/${params.SUITE}.xml"
            }
        }
        stage('Report') {
            steps {
                publishHTML(target: [
                    allowMissing: false,
                    alwaysLinkToLastBuild: true,
                    keepAll: true,
                    reportDir: 'target/extent-reports',
                    reportFiles: 'index.html',
                    reportName: 'Extent Report'
                ])
                junit 'target/surefire-reports/*.xml'
            }
        }
    }

    post {
        always {
            archiveArtifacts artifacts: 'target/**/*-report*.html, target/screenshots/**', allowEmptyArchive: true
            emailext(
                subject: "Build ${env.BUILD_NUMBER} - ${currentBuild.currentResult}",
                body: "Test run on ${params.ENVIRONMENT} with ${params.BROWSER}. See ${env.BUILD_URL}",
                to: 'qa-team@company.com'
            )
        }
        failure {
            slackSend(channel: '#automation-alerts',
                color: 'danger',
                message: "❌ Selenium build FAILED: ${env.JOB_NAME} #${env.BUILD_NUMBER}")
        }
        success {
            slackSend(channel: '#automation-alerts',
                color: 'good',
                message: "✅ Selenium build PASSED: ${env.JOB_NAME} #${env.BUILD_NUMBER}")
        }
    }
}

Scheduled Builds — Cron Syntax

triggers {
    cron('H 2 * * 1-5')  // Nightly at ~2 AM, Mon-Fri
    cron('0 6 * * 0')    // Full regression every Sunday 6 AM

Cron format: MINUTE HOUR DOM MONTH DOW. Use H for hash-based distribution to avoid load spikes.

import com.aventstack.extentreports.*;

public class ExtentReportManager {
    private static ExtentReports extent;
    private static ThreadLocal<ExtentTest> test = new ThreadLocal<>();

    public static ExtentReports getInstance() {
        if (extent == null) {
            ExtentSparkReporter spark = new ExtentSparkReporter("target/extent-reports/index.html");
            spark.config().setTheme(Theme.STANDARD);

            extent.setSystemInfo("OS", System.getProperty("os.name"));
            extent.setSystemInfo("Java", System.getProperty("java.version"));
            extent.setSystemInfo("Browser", System.getProperty("browser", "chrome"));
        }
        return extent;
    }

    public static void startTest(String name, String description) {
        ExtentTest extentTest = getInstance().createTest(name, description);
        test.set(extentTest);
    }

    public static ExtentTest getTest() { return test.get(); }

    public static void flush() { getInstance().flush(); }

Section 35: Selenium Grid

What Is Selenium Grid?

Selenium Grid enables distributed test execution across multiple machines and browsers simultaneously. A Hub receives test requests and routes them to registered Nodes that launch browsers — dramatically reducing total execution time for large regression suites.

Selenium Grid 4 Architecture

                    ┌──────────┐
   Test Script ────►│  Router  │ (Entry point :4444)
                    └────┬─────┘
                         │
              ┌──────────┼──────────┐
              ▼          ▼          ▼
        ┌──────────┐ ┌──────────┐ ┌──────────┐
        │Session   │ │Distributor│ │Event Bus │
        │Queue     │ │          │ │ (Redis)  │
        └──────────┘ └────┬─────┘ └──────────┘
                          │
              ┌───────────┼───────────┐
              ▼           ▼           ▼
        ┌──────────┐ ┌──────────┐ ┌──────────┐
        │ Node 1   │ │ Node 2   │ │ Node 3   │
        │ Chrome   │ │ Firefox  │ │ Edge     │
        └──────────┘ └──────────┘ └──────────┘

Components

  • Router: Single entry point receiving WebDriver requests
  • Distributor: Matches session requests to available nodes
  • Session Map: Tracks active sessions and their node assignments
  • Node: Machine running browsers; registers capabilities with hub
  • Event Bus: Internal communication channel between components

Section 36: Common Selenium Exceptions

Understanding Selenium exceptions is critical for debugging flaky tests and building robust frameworks. Each exception below includes cause, failing code, fix, and prevention strategy.

1. NoSuchElementException

What: Element not found in the DOM. When: Wrong locator, element not yet loaded, element inside iframe/shadow DOM.

// Code causing exception
WebElement btn = driver.findElement(By.id("submit-btn")); // ID doesn't exist

// Fix
WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));
WebElement btn = wait.until(ExpectedConditions.presenceOfElementLocated(By.id("submit-btn")));

// If in iframe:
driver.switchTo().frame("payment-frame");
WebElement btn = driver.findElement(By.id("submit-btn"));

Prevention: Verify locators in DevTools; use explicit waits; check iframe context.

2. TimeoutException

What: Wait condition not met within timeout period. When: Element never appeared, slow network, AJAX delay.

// Code causing exception
wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("loader-done"))); // 10s timeout exceeded

// Fix
WebDriverWait longWait = new WebDriverWait(driver, Duration.ofSeconds(30));
longWait.until(ExpectedConditions.invisibilityOfElementLocated(By.id("spinner")));
longWait.until(ExpectedConditions.visibilityOfElementLocated(By.id("content")));

Prevention: Wait for loader disappearance first; use appropriate timeout; verify element exists in DOM.

3. StaleElementReferenceException

What: Element reference is no longer attached to DOM. When: Page refresh, DOM update, SPA navigation, AJAX re-render.

// Code causing exception
WebElement row = driver.findElement(By.xpath("//tr[1]"));
driver.findElement(By.id("refresh")).click(); // DOM rebuilt
row.click(); // Stale!

// Fix — re-find element
driver.findElement(By.id("refresh")).click();
WebElement row = wait.until(ExpectedConditions.elementToBeClickable(By.xpath("//tr[1]")));
row.click();

Prevention: Re-locate elements after page changes; use dynamic locators instead of caching WebElement references.

4. ElementClickInterceptedException

What: Another element receives the click instead of target. When: Overlay, popup, sticky header/footer, loading spinner.

// Code causing exception
driver.findElement(By.id("checkout-btn")).click(); // Cookie banner overlay blocks click

// Fix options:
wait.until(ExpectedConditions.invisibilityOfElementLocated(By.id("cookie-banner")));
((JavascriptExecutor) driver).executeScript("arguments[0].click();", element);
new Actions(driver).moveToElement(element).click().perform();

Prevention: Close overlays before interaction; scroll element into view; wait for blocking elements to disappear.

5. ElementNotInteractableException

What: Element present in DOM but not visible/enabled. When: Hidden (display:none), off-screen, disabled, collapsed accordion.

// Code causing exception
driver.findElement(By.id("hidden-input")).sendKeys("text"); // display: none

// Fix
WebElement el = wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("hidden-input")));
((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView(true);", el);
el.sendKeys("text");

Prevention: Use visibilityOfElementLocated; scroll into view; expand collapsed sections first.

6. NoSuchWindowException

What: Target window/tab no longer exists. When: Window closed by script, popup auto-closed, wrong handle used.

// Code causing exception
driver.switchTo().window("invalid-handle-id");

// Fix
Set<String> handles = driver.getWindowHandles();
for (String handle : handles) {
    driver.switchTo().window(handle);
    if (driver.getTitle().contains("Expected Title")) break;
}

Prevention: Always verify handle exists before switching; store original window handle for return.

7. NoSuchFrameException

What: Specified frame/iframe not found. When: Wrong frame name/index, frame not yet loaded.

// Code causing exception
driver.switchTo().frame("nonexistent-frame");

// Fix
wait.until(ExpectedConditions.frameToBeAvailableAndSwitchToIt(By.id("payment-iframe")));
// Work inside frame
driver.switchTo().defaultContent(); // Always switch back

Prevention: Wait for frame availability; use ID/name over index; always return to defaultContent().

8. InvalidSelectorException

What: Malformed CSS or XPath locator syntax. When: Unclosed brackets, invalid characters, bad XPath axes.

// Code causing exception
driver.findElement(By.xpath("//input[@id='email'")); // Missing closing bracket

// Fix — validate in DevTools Console:
// document.querySelector("input#email")  or  $x("//input[@id='email']")
driver.findElement(By.xpath("//input[@id='email']"));

Prevention: Test locators in browser DevTools before using in code; use Page Object constants.

9. WebDriverException

What: Generic WebDriver failure — often browser/driver version mismatch. When: Incompatible ChromeDriver, driver not in PATH, browser not installed.

// Fix with WebDriverManager
WebDriverManager.chromedriver().setup();
WebDriver driver = new ChromeDriver();

// Or rely on Selenium Manager (built into Selenium 4.6+)
WebDriver driver = new ChromeDriver(); // Auto-downloads matching driver

Prevention: Use WebDriverManager or Selenium Manager; pin browser versions in CI Docker images.

10. SessionNotCreatedException

What: Browser session cannot be created. When: Driver/browser version mismatch, browser binary missing, port conflict, insufficient resources.

// Common causes and fixes:
// 1. Version mismatch → update driver or browser
WebDriverManager.chromedriver().clearDriverCache().setup();

// 2. Port conflict on Grid
// Change hub port: java -jar selenium-server.jar hub --port 4445

// 3. Headless in Docker without dependencies
ChromeOptions options = new ChromeOptions();
options.addArguments("--no-sandbox", "--disable-dev-shm-usage", "--headless=new");

Prevention: Align browser and driver versions; use official Docker images; check system resources.

Exception Handling Framework Strategy

public class ElementUtils {
    private WebDriver driver;
    private WebDriverWait wait;

    public WebElement findElementSafely(By locator) {
        try {
            return wait.until(ExpectedConditions.presenceOfElementLocated(locator));
        } catch (TimeoutException e) {
            captureScreenshot("element-not-found");
            throw new FrameworkException("Element not found: " + locator, e);
        } catch (NoSuchElementException e) {
            throw new FrameworkException("Invalid locator: " + locator, e);
        }
    }

    public void clickSafely(WebElement element) {
        try {
            wait.until(ExpectedConditions.elementToBeClickable(element)).click();
        } catch (ElementClickInterceptedException e) {
            ((JavascriptExecutor) driver).executeScript("arguments[0].click();", element);
        } catch (StaleElementReferenceException e) {
            // Re-find and retry once
            throw new FrameworkException("Stale element after retry", e);
        }
    }
}

public class FrameworkException extends RuntimeException {
    public FrameworkException(String message, Throwable cause) {
        super(message, cause);
    }
}

public class LoginPage {

    @FindBy(id = "password") private WebElement passwordField;
    @FindBy(id = "login-btn") private WebElement loginBtn;
    @FindBy(id = "sso-login") private WebElement ssoBtn;
    @FindBy(id = "otp-input") private WebElement otpField;

    private WebDriver driver;
    public LoginPage(WebDriver driver) {
        this.driver = driver;
        PageFactory.initElements(driver, this);
    }

    public DashboardPage loginWithEmailPassword(String email, String password) {
        emailField.sendKeys(email);
        passwordField.sendKeys(password);
        loginBtn.click();
        return new DashboardPage(driver);
    }

    public DashboardPage loginWithSSO(String provider) {
        ssoBtn.click();
        driver.findElement(By.xpath("//button[text()='" + provider + "']")).click();
        WaitUtils.waitForUrlContains(driver, "dashboard");
        return new DashboardPage(driver);
    }

    public DashboardPage loginWithOTP(String email, String password, String otp) {
        loginWithEmailPassword(email, password);
        otpField.sendKeys(otp);
        verifyOtpBtn.click();
        return new DashboardPage(driver);
    }

Scenario 5: Database Validation

Requirement: UI action triggers DB change; verify data integrity across UI, DB, and API layers.

DBUtils.java — JDBC Connection

public class DBUtils {
    private static Connection connection;

    public static Connection getConnection() throws SQLException {
        if (connection == null || connection.isClosed()) {
            connection = DriverManager.getConnection(
                ConfigReader.get("db.url"),
                ConfigReader.get("db.user"),
                ConfigReader.get("db.password")
            );
        }
        return connection;
    }

    public static Map<String, Object> getOrderById(String orderId) throws SQLException {
        String sql = "SELECT * FROM orders WHERE order_id = ?";
        PreparedStatement ps = getConnection().prepareStatement(sql);
        ps.setString(1, orderId);
        ResultSet rs = ps.executeQuery();
        Map<String, Object> row = new HashMap<>();
        if (rs.next()) {
            ResultSetMetaData meta = rs.getMetaData();
            for (int i = 1; i <= meta.getColumnCount(); i++) {
                row.put(meta.getColumnName(i), rs.getObject(i));
            }
        }
        return row;
    }

    public static int getUserCount() throws SQLException {
        ResultSet rs = getConnection().createStatement()
            .executeQuery("SELECT COUNT(*) FROM users");
        rs.next();
        return rs.getInt(1);
    }

    public static void updateOrderStatus(String orderId, String status) throws SQLException {
        PreparedStatement ps = getConnection().prepareStatement(
            "UPDATE orders SET status = ? WHERE order_id = ?");
        ps.setString(1, status);
        ps.setString(2, orderId);
        ps.executeUpdate();
    }
}

DriverFactory with ThreadLocal

public class DriverFactory {
    private static ThreadLocal<WebDriver> tlDriver = new ThreadLocal<>();

    public static WebDriver getDriver() {
        if (tlDriver.get() == null) {
            tlDriver.set(createDriver(System.getProperty("browser", "chrome")));
        }
        return tlDriver.get();
    }

    private static WebDriver createDriver(String browser) {
        return switch (browser.toLowerCase()) {
            case "firefox" -> new FirefoxDriver(new FirefoxOptions().addArguments("--start-maximized"));
            case "edge" -> new EdgeDriver(new EdgeOptions().addArguments("--start-maximized"));
            default -> new ChromeDriver(new ChromeOptions().addArguments("--start-maximized"));
        };
    }

    public static void quitDriver() {
        if (tlDriver.get() != null) {
            tlDriver.get().quit();
            tlDriver.remove();
        }
    }
}

BaseTest.java

public class BaseTest {
    protected WebDriver driver;

    @BeforeMethod
    @Parameters("browser")
    public void setUp(@Optional("chrome") String browser) {
        System.setProperty("browser", browser);
        driver = DriverFactory.getDriver();
        driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(10));
    }

    @AfterMethod
    public void tearDown() {
        DriverFactory.quitDriver();
    }
