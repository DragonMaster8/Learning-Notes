package com.automation.pages;

import org.openqa.selenium.*;
import org.openqa.selenium.support.ui.*;

public class LoginPage {
    private WebDriver driver;
    private WebDriverWait wait;

    private By emailField    = By.id("email");
    private By passwordField = By.id("password");
    private By loginBtn      = By.cssSelector("button[type='submit']");
    private By errorMsg      = By.cssSelector(".error-message");
    private By infoMsg       = By.cssSelector(".info-message");
    private By otpTab        = By.id("otp-tab");
    private By mobileField   = By.id("mobile");
    private By sendOtpBtn    = By.id("send-otp");
    private By otpField      = By.id("otp-input");
    private By verifyOtpBtn  = By.id("verify-otp");

    public LoginPage(WebDriver driver) {
        this.driver = driver;
        this.wait = new WebDriverWait(driver, Duration.ofSeconds(15));
    }

    public void enterEmail(String email) {
        wait.until(ExpectedConditions.visibilityOfElementLocated(emailField));
        driver.findElement(emailField).clear();
        driver.findElement(emailField).sendKeys(email);
    }

    public void enterPassword(String password) {
        driver.findElement(passwordField).clear();
        driver.findElement(passwordField).sendKeys(password);
    }

    public void clickLoginButton() {
        driver.findElement(loginBtn).click();
    }

    public void clickSSOProvider(String provider) {
        driver.findElement(By.cssSelector("[data-provider='" + provider.toLowerCase() + "']")).click();
    }

    public void handleSSOPopup(String provider, String user, String pass) {
        String mainWindow = driver.getWindowHandle();
        wait.until(d -> d.getWindowHandles().size() > 1);
        for (String handle : driver.getWindowHandles()) {
            if (!handle.equals(mainWindow)) {
                driver.switchTo().window(handle);
                break;
            }
        }
        driver.findElement(By.id("identifierId")).sendKeys(user);
        driver.findElement(By.id("identifierNext")).click();
        wait.until(ExpectedConditions.visibilityOfElementLocated(By.name("Passwd")));
        driver.findElement(By.name("Passwd")).sendKeys(pass);
        driver.findElement(By.id("passwordNext")).click();
        driver.switchTo().window(mainWindow);
    }

    public void cancelSSOPopup() {
        String mainWindow = driver.getWindowHandle();
        wait.until(d -> d.getWindowHandles().size() > 1);
        for (String handle : driver.getWindowHandles()) {
            if (!handle.equals(mainWindow)) {
                driver.switchTo().window(handle).close();
                break;
            }
        }
        driver.switchTo().window(mainWindow);
    }

    public void switchToOTPTab() { driver.findElement(otpTab).click(); }
    public void enterMobile(String mobile) { driver.findElement(mobileField).sendKeys(mobile); }
    public void clickSendOTP() { driver.findElement(sendOtpBtn).click(); }
    public void enterOTP(String otp) { driver.findElement(otpField).sendKeys(otp); }
    public void clickVerifyOTP() { driver.findElement(verifyOtpBtn).click(); }

    public String getErrorMessage() {
        return wait.until(ExpectedConditions.visibilityOfElementLocated(errorMsg)).getText();
    }

    public String getInfoMessage() {
        return wait.until(ExpectedConditions.visibilityOfElementLocated(infoMsg)).getText();
    }
}

package com.automation.api.steps;

import io.cucumber.java.en.*;
import io.restassured.response.Response;
import io.restassured.specification.RequestSpecification;
import org.testng.Assert;
import static io.restassured.RestAssured.*;
import static org.hamcrest.Matchers.*;

public class UserApiSteps {
    private RequestSpecification request;
    private Response response;
    private String userId;
    private long startTime;

    @Given("the API base URL is {string}")
    public void setBaseUrl(String url) {
        baseURI = url;
        request = given().header("Content-Type", "application/json");
    }

    @Given("the API key is configured")
    public void setApiKey() {
        request = request.header("X-API-Key", System.getProperty("api.key", "test-key-123"));
    }

    @Given("a user exists with email {string}")
    public void createUser(String email) {
        response = request.body("{\"name\":\"Jane Doe\",\"email\":\"" + email + "\",\"role\":\"member\"}")
                          .post("/users");
        userId = response.jsonPath().getString("id");
    }

    @When("I send POST request to {string} with body:")
    public void postRequest(String endpoint, String body) {
        startTime = System.currentTimeMillis();
        response = request.body(body).post(endpoint);
    }

    @When("I send GET request to {string}")
    public void getRequest(String endpoint) {
        startTime = System.currentTimeMillis();
        String path = endpoint.replace("{userId}", userId);
        response = request.get(path);
    }

    @When("I send PUT request to {string} with body:")
    public void putRequest(String endpoint, String body) {
        String path = endpoint.replace("{userId}", userId);
        response = request.body(body).put(path);
    }

    @When("I send DELETE request to {string}")
    public void deleteRequest(String endpoint) {
        String path = endpoint.replace("{userId}", userId);
        response = request.delete(path);
    }

    @Then("the response status code should be {int}")
    public void verifyStatus(int code) {
        Assert.assertEquals(response.getStatusCode(), code);
    }

    @Then("the response should contain field {string}")
    public void verifyFieldExists(String field) {
        Assert.assertNotNull(response.jsonPath().get(field));
        userId = response.jsonPath().getString("id");
    }

    @Then("the response field {string} should be {string}")
    public void verifyFieldValue(String field, String expected) {
        Assert.assertEquals(response.jsonPath().getString(field), expected);
    }

    @Then("response time should be less than {int} ms")
    public void verifyResponseTime(int maxMs) {
        long elapsed = System.currentTimeMillis() - startTime;
        Assert.assertTrue(elapsed < maxMs, "Response took " + elapsed + "ms");
    }

    @Then("the response should contain error message {string}")
    public void verifyErrorMessage(String msg) {
        Assert.assertTrue(response.jsonPath().getString("error").contains(msg));
    }
}

package com.automation.steps;

import com.automation.pages.EmployeePage;
import com.automation.utils.*;
import io.cucumber.java.en.*;
import org.testng.Assert;
import java.sql.*;
import java.util.*;

public class DatabaseValidationSteps {
    private Connection conn;
    private EmployeePage empPage;
    private List<Map<String, String>> uiEmployees;

    @Given("database connection is established")
    public void connectDB() throws SQLException {
        conn = DBUtil.getConnection(
            ConfigReader.get("db.url"),
            ConfigReader.get("db.user"),
            ConfigReader.get("db.password"));
        Assert.assertFalse(conn.isClosed());
    }

    @Given("I am on the employee management page")
    public void openEmployeePage() {
        DriverFactory.getDriver().get(ConfigReader.get("app.url") + "/employees");
        empPage = new EmployeePage(DriverFactory.getDriver());
    }

    @When("I fetch all employees from UI")
    public void fetchUIEmployees() {
        uiEmployees = empPage.getAllEmployees();
    }

    @Then("UI employee count should match database count")
    public void compareCount() throws SQLException {
        int dbCount = DBUtil.queryCount(conn, "SELECT COUNT(*) FROM employees WHERE active = 1");
        Assert.assertEquals(uiEmployees.size(), dbCount);
    }

    @Then("each UI employee should match database record")
    public void compareRecords() throws SQLException {
        for (Map<String, String> uiEmp : uiEmployees) {
            Map<String, Object> dbRow = DBUtil.querySingle(conn,
                "SELECT name, department, salary FROM employees WHERE id = ?",
                uiEmp.get("id"));
            Assert.assertEquals(uiEmp.get("name"), dbRow.get("name"));
            Assert.assertEquals(uiEmp.get("department"), dbRow.get("department"));
        }
    }

    @When("I create employee with name {string} department {string} salary {string}")
    public void createEmployee(String name, String dept, String salary) {
        empPage.clickAddEmployee();
        empPage.fillEmployeeForm(name, dept, salary);
        empPage.saveEmployee();
    }

    @Then("employee {string} should exist in database")
    public void verifyExists(String name) throws SQLException {
        Map<String, Object> row = DBUtil.querySingle(conn,
            "SELECT id FROM employees WHERE name = ?", name);
        Assert.assertNotNull(row, "Employee not found in DB: " + name);
    }

    @Then("database field {string} for {string} should be {string}")
    public void verifyDBField(String field, String name, String expected) throws SQLException {
        Map<String, Object> row = DBUtil.querySingle(conn,
            "SELECT " + field + " FROM employees WHERE name = ?", name);
        Assert.assertEquals(String.valueOf(row.get(field)), expected);
    }

    @Given("employee {string} exists in database with department {string}")
    public void seedEmployee(String name, String dept) throws SQLException {
        DBUtil.execute(conn,
            "INSERT INTO employees (name, department, salary, active) VALUES (?, ?, 60000, 1)",
            name, dept);
        empPage.refreshPage();
    }

    @When("I update employee {string} department to {string}")
    public void updateEmployee(String name, String dept) {
        empPage.editEmployee(name);
        empPage.setDepartment(dept);
        empPage.saveEmployee();
    }

    @When("I delete employee {string} from UI")
    public void deleteEmployee(String name) {
        empPage.deleteEmployee(name);
        empPage.confirmDelete();
    }

    @Then("employee {string} should NOT exist in database")
    public void verifyNotExists(String name) throws SQLException {
        Map<String, Object> row = DBUtil.querySingle(conn,
            "SELECT id FROM employees WHERE name = ?", name);
        Assert.assertNull(row);
    }
}

package com.automation.utils;

import java.io.*;
import java.util.Properties;

public class ConfigReader {
    private static Properties props = new Properties();
    static {
        try (InputStream is = ConfigReader.class.getClassLoader()
                .getResourceAsStream("config.properties")) {
            props.load(is);
        } catch (IOException e) { throw new RuntimeException(e); }
    }
    public static String get(String key) {
        String sysVal = System.getProperty(key);
        return sysVal != null ? sysVal : props.getProperty(key);
    }
    public static String getBrowser() {
        return get("browser"); // set via -Dbrowser=chrome or testng parameter
    }
}

// DriverFactory.java — browser selected from ConfigReader
public class DriverFactory {
    private static ThreadLocal<WebDriver> tlDriver = new ThreadLocal<>();

    public static WebDriver getDriver() {
        if (tlDriver.get() == null) {
            String browser = ConfigReader.getBrowser();
            switch (browser.toLowerCase()) {
                case "firefox":
                    tlDriver.set(new FirefoxDriver());
                    break;
                default:
                    tlDriver.set(new ChromeDriver());
            }
            tlDriver.get().manage().window().maximize();
        }
        return tlDriver.get();
    }

    public static void quitDriver() {
        if (tlDriver.get() != null) {
            tlDriver.get().quit();
            tlDriver.remove();
        }
    }
}

public class RetryAnalyzer implements IRetryAnalyzer {
    private int count = 0;
    private static final int MAX_RETRY = 2;

    @Override
    public boolean retry(ITestResult result) {
        if (count < MAX_RETRY) {

Top 100 TestNG Questions

Basic (30)

Q1: What is TestNG and why use it over JUnit?

Answer: TestNG (Test Next Generation) is a Java testing framework inspired by JUnit and NUnit. It provides advanced features like parallel execution, data providers, grouping, dependency testing, and flexible XML-based suite configuration. TestNG supports annotations for fine-grained lifecycle control (@BeforeSuite through @AfterMethod). It integrates seamlessly with Selenium, Maven, and CI/CD pipelines. Reporting is richer with default HTML reports and listener extensibility.

Interviewer expects: Clear comparison showing you chose TestNG for enterprise automation needs, not just naming features.

Q2: What annotations does TestNG provide for test lifecycle?

Answer: TestNG provides @BeforeSuite, @AfterSuite, @BeforeTest, @AfterTest, @BeforeClass, @AfterClass, @BeforeMethod, @AfterMethod, and @BeforeGroups/@AfterGroups. @BeforeSuite runs once before all tests; @BeforeMethod runs before each @Test. Execution order: Suite → Test → Class → Method. These annotations replace setup/teardown boilerplate and enable layered test initialization for UI, API, and DB layers independently.

Interviewer expects: Correct execution order and knowing which annotation runs at which scope.

Q3: What is the execution order of TestNG annotations?

Answer: Order: @BeforeSuite → @BeforeTest → @BeforeClass → @BeforeMethod → @Test → @AfterMethod → @AfterClass → @AfterTest → @AfterSuite. @BeforeGroups runs before the first method in matching groups; @AfterGroups runs after the last. Multiple @Test methods in a class each trigger @BeforeMethod/@AfterMethod pair. Priority attribute affects @Test order within a class but not lifecycle annotations.

Interviewer expects: Precise ordering without confusing @BeforeTest with @BeforeMethod scope.

Q4: How do you create a basic TestNG test?

Answer: Create a Java class, import org.testng.annotations.Test and org.testng.Assert. Annotate test methods with @Test. Use Assert.assertEquals, assertTrue, etc. for validations. Run via IDE (right-click → TestNG), Maven surefire plugin, or testng.xml suite. Example: @Test public void addTest() { Assert.assertEquals(2+2, 4); }. No main method needed; TestNG discovers @Test methods automatically.

Interviewer expects: Working syntax and awareness of multiple execution methods.

Q5: What is testng.xml and why is it used?

Answer: testng.xml is a TestNG suite configuration file written in XML. It defines which tests to run, groups to include/exclude, parallel execution settings, listeners, and parameters. It allows running subsets of tests without recompiling code. QA teams use it to create smoke, regression, and sanity suites. CI/CD pipelines reference testng.xml for consistent test selection across environments.

Interviewer expects: Understanding that XML decouples test selection from code — critical for CI pipelines.

Q6: What is the difference between @BeforeClass and @BeforeMethod?

Answer: @BeforeClass runs once before all test methods in a class; @BeforeMethod runs before each @Test method. Use @BeforeClass for expensive one-time setup like WebDriver initialization in non-parallel runs. Use @BeforeMethod for per-test setup like clearing cookies, navigating to base URL, or resetting test data. In parallel method execution, @BeforeClass is shared across threads — a common pitfall.

Interviewer expects: Practical guidance on when to use each, including parallel execution caveats.

Q7: What is the @Test annotation?

Answer: @Test marks a method as a test case executable by TestNG. Attributes include priority, groups, dependsOnMethods, enabled, timeOut, dataProvider, expectedExceptions, and retryAnalyzer. Without @Test, TestNG ignores the method. Multiple @Test methods can exist in one class. TestNG reports pass/fail/skip status per @Test method in HTML and XML reports.

Interviewer expects: Knowledge of key attributes, not just "it marks a test."

Q8: What assertion methods does TestNG provide?

Answer: TestNG provides assertEquals, assertNotEquals, assertTrue, assertFalse, assertNull, assertNotNull, assertSame, assertNotSame, and fail(). All are static methods in org.testng.Assert. assertEquals handles objects, strings, and primitives with optional message parameter. SoftAssert (org.testng.asserts.SoftAssert) collects multiple failures and reports them via assertAll() — useful for form validation with many fields.

Interviewer expects: Mention SoftAssert as differentiator; know assertEquals vs assertSame distinction.

Q9: What is the difference between hard assert and soft assert?

Answer: Hard assert (Assert.assertEquals) stops test execution immediately on failure. Soft assert (SoftAssert) continues executing subsequent assertions and reports all failures at assertAll(). Use hard assert for critical checkpoints (login succeeded). Use soft assert for non-blocking validations (multiple UI field checks). SoftAssert must be instantiated per test method; never use static SoftAssert in parallel tests.

Interviewer expects: Real-world usage scenario, not just definition.

Q10: What are TestNG groups and how do you use them?

Answer: Groups categorize tests for selective execution. Assign via @Test(groups = {"smoke", "regression"}). In testng.xml, use <include> and <exclude> under <groups><run> to filter. A test can belong to multiple groups. Groups enable smoke runs on every commit and full regression nightly. Maven: mvn test -Dgroups=smoke. Groups differ from Cucumber tags but serve the same filtering purpose.

Interviewer expects: XML syntax fluency and CI integration use case.

Q11: How do you skip a test in TestNG?

Answer: Set @Test(enabled = false) to skip a test programmatically. In testng.xml, exclude groups or specific classes. Use dependsOnMethods — if dependency fails, dependent tests skip automatically. ITestListener.onTestSkipped captures skipped tests in custom reports. Skipped ≠ Failed: skipped tests don't indicate defects. Use skip for known broken features tracked in Jira.

Interviewer expects: Distinguish skip vs fail and know multiple skip mechanisms.

Q12: What is test priority in TestNG?

Answer: @Test(priority = 1) controls execution order within a class. Lower numbers run first. Default priority is 0. Priority only affects methods in the same class — not across classes. Unlike dependsOnMethods, priority doesn't skip tests on failure. Use priority for logical ordering (create before read). Best practice: prefer dependsOnMethods for true dependencies, priority for ordering convenience.

Interviewer expects: Know priority limitations and not confuse with dependsOnMethods.

Q13: What is dependsOnMethods in TestNG?

Answer: @Test(dependsOnMethods = {"loginTest"}) ensures the current test runs only if loginTest passes. If dependency fails, TestNG marks this test as SKIPPED. Can depend on multiple methods: dependsOnMethods = {"a", "b"}. Dependencies work within the same class by default. Cross-class dependencies require dependsOnGroups or careful suite ordering. Overusing dependencies creates brittle test chains — prefer independent tests.

Interviewer expects: Understand skip behavior and warn against over-coupling tests.

Q14: How do you run TestNG tests from command line?

Answer: Via Maven: mvn clean test (uses surefire plugin). Specify suite: mvn test -DsuiteXmlFile=testng.xml. Run single class: mvn test -Dtest=LoginTest. Direct TestNG: java -cp "testng.jar:classes" org.testng.TestNG testng.xml. IDE: right-click class or XML → Run As TestNG Suite. CI: Jenkins executes mvn test with suite parameter per environment.

Interviewer expects: Maven command fluency — most teams use Maven, not raw java -cp.

Q15: What is the @Parameters annotation?

Answer: @Parameters({"browser", "url"}) injects values from testng.xml <parameter> tags into test methods. XML: <parameter name="browser" value="chrome"/>. Method: public void test(@Optional("chrome") String browser). @Optional provides default if parameter missing. Useful for cross-browser testing and environment URLs. Limitation: only supports String parameters — use @DataProvider for complex types.

Q3: What is ubiquitous language?

Answer: TDD: developer writes unit tests first, then code (Red-Green-Refactor). BDD: team defines behavior examples first using Gherkin, then automates. TDD focuses on code correctness at unit level. 18Cucumber + TestNG integration?Runner extends AbstractTestNGCucumberTests; testng.xml runs runner.
19Doc Strings?Multi-line text block passed as String parameter to step definition.
20Scenario Context?Scenario object in hooks provides name, tags, status, line number.

22PicoContainer in Cucumber?Dependency injection sharing step definition instances across steps.
